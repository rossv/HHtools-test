packages = ["pandas", "numpy"] paths = ["./hh_tools"] import js from pyodide.ffi import create_proxy import pandas as pd from hh_tools.design_storm import build_storm def generate(event): depth = float(js.document.getElementById('depth').value) duration = float(js.document.getElementById('duration').value) timestep = float(js.document.getElementById('timestep').value) dist = js.document.getElementById('distribution').value df = build_storm(depth=depth, duration_hr=duration, timestep_min=timestep, distribution=dist) js.document.getElementById('output').innerHTML = df.to_html(classes='table', index=False) js.document.getElementById('generate').addEventListener('click', create_proxy(generate))
